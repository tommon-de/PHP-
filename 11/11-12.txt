    foreach($list1 as $k)

    $list2=scandir($path2);//读取第二个路径

    echo "移动前的".$path2."内容列表<br>";
    foreach($list2 as $k)
        {echo $k."；";}
    rename($path1.'\images',$path2.'\newdir');  //移动文件夹
    echo "<br>";
    $list1=scandir($path1); //读取第一个路径
    echo "移动后的".$path1."内容列表<br>";
    foreach($list1 as $k)
        {echo $k."；";}
    echo "<br>";
    $list2=scandir($path2); //读取第二个路径

        {echo $k."；";}
